1.介绍linux

2 命令
	cd 	跳转路径
		绝对路径
		相对路径
		
		..	上一级目录
		.	当前目录
		-	上一次所在目录
		~	home目录
		/	根目录
		
	ls 列出当前文件夹下的文件
		-a	显示隐藏
		-l	列表方式
		-h	以k方式显示大小
		
	>	重定向
	>>	重定向（追加）
	
	pwd	查看当前所在路径
	
	touch	创建文件
	
	mkdir	创建文件夹
	
	rmdir	删除空文件夹
	
	rm	删除（可删非空，文件，文件夹）
		-rf	强制删除
		
	|	管道
	
	more	分屏显示
	less	分屏显示
	
	cp	拷贝
	mv	移动
		mv [旧名字] [新名字]	重命名
	
	--help	查看帮助文档
	
	man	（manual）手册
	
	tab	自动补全
	
	ps	查看当前系统运行的进程
		-aux
		
	top	动态显示当前进程占用系统的信息
	
	kill	关闭进程
		-9	强制关闭进程	kill PID -9
	
	clear	清屏
	
	cat	查看文件
	
	date	显示时间
	
	cay	日历
		-y	显示当年
	
	df	查看磁盘使用情况
	
	du	查看当前目录中所用文件及文件夹的占用情况
	
	shutdown	关机
		-r new	重启
		-h new	立即关机
		-h 20:55	到20:55关机
		-h +10	十分钟后关机
		
	halt	关机
	
	init 0	关机
	init 6	重启
	
	ping
	
	ifconfig
